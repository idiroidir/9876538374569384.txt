function decode_phizzle($key, $chunks) {
    $payload = implode('', $chunks);
    $data = base64_decode($payload);
    
    $decrypted = '';
    for ($i = 0; $i < strlen($data); $i++) {
        $decrypted .= $data[$i] ^ $key[$i % strlen($key)];
    }
    
    // Fix 3: Use gzuncompress instead of gzinflate
    return gzuncompress($decrypted);
}
